2. push_back = The push_back function adds a new element at the end of the vector.
//output 1,2,3,4,5

    v.push_back(4);
    v.push_back(5);
    for (int i : v) {
        cout << i << " ";
}
    return 0;
}

5. iterator = Vectors support iterators, which allow traversal through the elements in the vector. Iterators behave like pointers and can be used for accessing elements or performing operations on the vector.
//output = 10,20,30

#include <bits/stdc++.h>
using namespace std;
        
int main() {
vector<int> v = {10, 20, 30};
for (vector<int>::iterator it = v.begin(); it != v.end(); ++it) {
    cout << *it << " ";
}
return 0;
}

7. erase The erase function removes an element from the vector by a position causing all subsequent elements to be shifted one position to the left.
//output = 10,30

#include <bits/stdc++.h>
using namespace std;
        
int main() {
    vector<int> v = {10, 20, 30};
    v.erase(v.begin() + 1);
    for (int i : v) {
        cout << i << " ";
}
        return 0;
}

11. swap = The swap function swaps the contents of two vectors. 
//output = 4,5,6

#include <bits/stdc++.h>
using namespace std;
        
int main() {
    vector<int> v1 = {1, 2, 3};
    vector<int> v2 = {4, 5, 6};
    swap(v1, v2);
    for (int i : v1) {
        cout << i << " ";
    }
    return 0;
}

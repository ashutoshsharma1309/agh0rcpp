-> vector tasks

1. push_back() functions adds a new element at the end of the vector

#include <bits/stdc++.h>
using namespace std;
        
int main() {
    vector<int> v = {1, 2, 3};
    v.push_back(4);
    v.push_back(5);//O(1)
    for (int i : v) {
        cout << i << " ";
    }
    return 0;
}

2. pop_back() function pops a element at the end of the vector

#include <bits/stdc++.h>
using namespace std;
        
int main() {
    vector<int> v = {1, 2, 3};
    v.pop_back(3);
    v.pop_back(2);//0(1)
    for (int i : v) {
        cout << i << " ";
    }
    return 0;
}

#include <iostream>
using namespace std;

int main(){
    vector <int> v = {2,3,5,6,7};
    for(int i = 0; i < v.size(); ++i){
        cout << v[i] << " ";
    }
    cout << endl;
    vector<int> ::iterator it = v.begin();
    cout << (*(it+1)) << endl;
